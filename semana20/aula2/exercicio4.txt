exports.handler = async event => {
    if(event.email.indexOf('@') === -1){
        return {
            "isEmail": false,
            "reason": "Não possui '@'"
        }
    }

    if(event.email.indexOf('.') === -1){
        return {
            "isEmail": false,
            "reason": "Não possui '.'"
        }
    }

    return {
        "isEmail": true
    }
};
